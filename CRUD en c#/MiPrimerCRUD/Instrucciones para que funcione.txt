1. CREAR LA BASE DE DATOS DESDE EL VISUAL Y CONECTARLA EN EL EXPLORADOR DE SERVIDORES
Y ORIGENES DE DATOS

2. CREAR LAS TABLAS:

CREATE TABLE [dbo].[Articulo] (
    [Id]                INT           IDENTITY (1, 1) NOT NULL,
    [Seccion]           NVARCHAR (50) NULL,
    [NombreDelArticulo] NVARCHAR (50) NULL,
    [Precio]            DECIMAL (18)  NULL,
    [Fecha]             DATETIME      DEFAULT (getdate()) NULL,
    [PaisDeOrigen]      NVARCHAR (50) NULL,
    [Proveedor]         NVARCHAR (50) NULL,
    [Cantidad]          INT           NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[Categoria] (
    [Id]     INT           IDENTITY (1, 1) NOT NULL,
    [Nombre] NVARCHAR (50) NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);


CREATE TABLE [dbo].[Cliente] (
    [Id]        INT           IDENTITY (1, 1) NOT NULL,
    [Nombre]    NVARCHAR (50) NULL,
    [Direccion] NVARCHAR (50) NULL,
    [Poblacion] NVARCHAR (50) NULL,
    [Telefono]  NVARCHAR (50) NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[Proveedor] (
    [Id]        INT           IDENTITY (1, 1) NOT NULL,
    [Nombre]    NVARCHAR (50) NULL,
    [Telefono]  NVARCHAR (50) NULL,
    [Direccion] NVARCHAR (50) NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[Venta] (
    [Id]                  INT           IDENTITY (1, 1) NOT NULL,
    [CodigoDelCliente]    INT           NOT NULL,
    [NombreDelCliente]    NVARCHAR (50) NULL,
    [FechaDelPedido]      DATETIME      DEFAULT (getdate()) NULL,
    [FormaDePago]         NVARCHAR (50) NULL,
    [NombreDelProducto]   NVARCHAR (50) NULL,
    [Monto]               DECIMAL (18)  NULL,
    [CantidadDelProducto] INT           NULL,
    [SubTotal]            DECIMAL (18)  NULL,
    [PrecioDelProducto]   DECIMAL (18)  NULL,
    [MontoDeCambio]       DECIMAL (18)  NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_Venta_ToTable] FOREIGN KEY ([CodigoDelCliente]) REFERENCES [dbo].[Cliente] ([Id])
);
